# RTCSctpTransport internals (_outbound_streams_count, _inbound_streams_max)
# that may move between minor versions. 1.14 is the version validated against
# the live camera.

# ConnectionClosed + the ping/close_timeout kwargs) is identical on the >=14
# asyncio implementation — verified against 16.x. So the <14 cap is no longer
# needed; allow the modern line (the container was already resolving 16.x).
websockets>=12.0,<17.0
av>=14.0.0,<17.0.0
numpy>=2.4.6,<3.0
